FOR (a:Author) REQUIRE a.email IS UNIQUE;

CREATE CONSTRAINT unique_article_doi IF NOT EXISTS
FOR (ar:Article) REQUIRE ar.doi IS UNIQUE;

CREATE CONSTRAINT unique_topic_name IF NOT EXISTS
